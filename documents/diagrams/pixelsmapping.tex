\begin{tikzpicture}
    \draw[axis] (-4, 0) -- (4, 0);
    \draw[axis] (0, -4) -- (0, 4);

    \filldraw (0, 0) circle (1pt);
    \draw (-4, -4) grid (4, 4);

    \foreach \i in {-4, -3,..., 4} {
        \foreach \j in {-4, -3,..., 4} {
            \filldraw (\i, \j) circle (3pt);
            \tikzmath{
                \num = \i + 5 - 9 * (\j - 4);
            }
            \node[below right] at (\i, \j){\pgfmathprintnumber[precision=0]{\num}};
        }
    }
\end{tikzpicture}
